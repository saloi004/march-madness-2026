# 1. 导入必备的数据清洗和建模包
# 如果没安装，请先运行 
install.packages(c("dplyr", "readr", "caret"))
library(dplyr)
library(readr)
# ==========================================
# 第一步：读取数据
# ==========================================
# 1. 读取你上传的 2026 年待预测对阵表
matchups_2026 <- read_csv("2026_Potential_Matchups.csv")

# 2. 读取球队年度统计数据 (使用你在Kaggle下载的数据集)
# 假设包含 TeamID, 场均得分(Points), 进攻效率(Off_Rtg) 等
# 文件名请替换为你实际解压出来的文件，例如 "Team_Stats_2026.csv"
team_stats <- read_csv("Team_Stats.csv") 

# 3. 读取历史锦标赛的比赛结果 (用来训练电脑)
# 假设包含过去的对阵以及谁赢了，例如 "Historical_Tournament_Matchups.csv"
historical_matchups <- read_csv("Historical_Tournament_Matchups.csv")


# ==========================================
# 第二步：特征工程 (Feature Engineering)
# 核心思想：比赛的结果往往取决于两队的"实力差"
# ==========================================

# 我们写一个函数，把两支球队的统计数据拼接到对阵表里，并计算差值
prepare_features <- function(matchup_df, stats_df) {
  matchup_df %>%
    # A. 拼接待预测的 HigherSeed（高位种子队）的数据
    left_join(stats_df, by = c("HigherSeedID" = "TeamID")) %>%
    rename_with(~paste0("High_", .), -c(colnames(matchup_df))) %>%
    
    # B. 拼接待预测的 LowerSeed（低位种子队）的数据
    left_join(stats_df, by = c("LowerSeedID" = "TeamID")) %>%
    rename_with(~paste0("Low_", .), -c(colnames(matchup_df), starts_with("High_"))) %>%
    
    # C. 计算两队之间的差值（这里是模型预测的核心依据！）
    mutate(
      Seed_Diff = HigherSeedNum - LowerSeedNum, # 种子排位差
      
      # ⚠️ 注意：以下列名你需要根据 Kaggle 数据集里实际的列名进行修改
      # 比如：高位种子队的得分能力 减去 低位种子队的得分能力
      # Score_Diff = High_Points - Low_Points, 
      # Efficiency_Diff = High_Off_Rtg - Low_Off_Rtg
    )
}

# 应用这个函数处理我们的历史训练集和2026预测集
train_data <- prepare_features(historical_matchups, team_stats)
test_data  <- prepare_features(matchups_2026, team_stats)

# ==========================================
# 第三步：训练模型 (Train the Model)
# ==========================================

# 假设 train_data 中有一个叫 "HigherSeed_Won" 的列，如果是 Higher赢了就是 1，输了就是 0
# 我们使用逻辑回归 (glm, family = "binomial") 来预测
# 公式 ~ 后面放的就是你要用的特征变量
model <- glm(HigherSeed_Won ~ Seed_Diff,  # + Score_Diff + Efficiency_Diff (把你算出的特征加上去)
             data = train_data, 
             family = "binomial")

# 查看模型效果（看各个特征的 P-value 是否显著）
summary(model)


# ==========================================
# 第四步：输出预测结果 (Make Predictions)
# ==========================================

# 让模型对 matchups_2026 这个新表打分，type = "response" 确保输出的是百分比概率
matchups_2026$Predictions <- predict(model, newdata = test_data, type = "response")

# ==========================================
# 第五步：导出文件，准备提交！
# ==========================================

# 确保把所有可能产生 NA 的概率填补一下（比如用0.5盲猜，防止报错）
matchups_2026$Predictions[is.na(matchups_2026$Predictions)] <- 0.5

# 仅保留要求的列并导出为 CSV
matchups_2026 %>%
  select(HigherSeed, HigherSeedID, HigherSeedNum, LowerSeed, LowerSeedID, LowerSeedNum, Predictions) %>%
  write_csv("2026_Predictions_Submission.csv")

print("预测完成！已生成 2026_Predictions_Submission.csv")
